1. kkma 사용하기
``` python
# ───────────────────────────────────────────────
# [1] 필요한 도구 불러오기
# ───────────────────────────────────────────────

import re
# re = "regular expression(정규표현식)" 도구.
# 글자 속에서 특정 패턴(숫자, 영어 등)을 찾거나 바꿀 때 사용.

from collections import Counter
# Counter = 리스트 안에 있는 항목들이 각각 몇 번 나왔는지
# 자동으로 세어주는 도구. 빈도 분석에 딱 맞음.


# ───────────────────────────────────────────────
# [2] 지우고 싶은 글자 패턴 미리 정의하기
# ───────────────────────────────────────────────

pattern = re.compile(r'[0-9a-zA-Z\W]+|[ㅋㅎㅍㅇ]+')
# re.compile(...) = 패턴을 미리 만들어서 'pattern'이라는 이름에 저장.
#   (미리 만들어두면 여러 번 쓸 때 더 빠름)
#
# 패턴 내용을 쪼개서 보면:
#   [0-9]        → 숫자 0부터 9까지
#   [a-zA-Z]     → 영어 소문자 + 대문자
#   \W           → 글자/숫자가 아닌 것 = 공백, 마침표, 느낌표, 특수문자 등
#   +            → 위 글자들이 1개 이상 연속될 때
#   |            → "또는" (왼쪽 패턴 또는 오른쪽 패턴)
#   [ㅋㅎㅍㅇ]+  → ㅋ,ㅎ,ㅍ,ㅇ 같은 자음만 있는 글자 (ㅋㅋ, ㅎㅎ 등)
#
# 정리: 숫자 / 영어 / 특수문자·공백 / 의미없는 자음 → 이것들을 잡아내는 그물.


# ───────────────────────────────────────────────
# [3] 리뷰에서 단어를 꺼내 깨끗하게 청소하기
# ───────────────────────────────────────────────

clean_tokens = [
    cleaned
    # ↑ 최종적으로 리스트에 담을 값 (청소가 끝난 단어)
    #   ※ 아래 if에서 만들어진 cleaned를 여기서 사용함

    for doc in deliver_df['리뷰']
    # ↑ '리뷰' 열에서 리뷰 글을 하나씩 꺼냄 → doc 에 담음
    #   (doc = document, 리뷰 글 한 개)

    for token in doc.split()
    # ↑ 꺼낸 리뷰 글(doc)을 공백 기준으로 쪼개서 단어 하나씩 꺼냄 → token 에 담음
    #   예: "배송 정말 빠름" → "배송", "정말", "빠름"

    if (cleaned := pattern.sub('', token))
    # ↑ 핵심 줄. 천천히 보면:
    #   pattern.sub('', token) = token에서 위 패턴에 걸리는 글자를 ''(아무것도없음)으로
    #                            바꿈 = 즉 숫자·영어·특수문자·자음을 전부 지움.
    #   (cleaned := ...)        = 지운 결과를 'cleaned'에 저장하면서 동시에 그 값을 내놓음.
    #   if ...                  = 내놓은 값이 빈 문자열("")이면 거짓 → 버림.
    #                             글자가 남아 있으면 참 → 위의 cleaned가 리스트에 들어감.
    #
    #   예시:
    #     "배송3" → 숫자 3 지움 → "배송"  → 글자 있음 → 채택 ✅
    #     "ㅋㅋ"  → 전부 지움   → ""      → 빈칸     → 버림 ❌
]
# 결과: clean_tokens = ['배송', '정말', '빠름', ...] 처럼 청소된 단어들의 리스트.


# ───────────────────────────────────────────────
# [4] 형태소 분석 (단어를 쪼개고 품사를 붙이기)
# ───────────────────────────────────────────────

dev = kkma.pos(" ".join(clean_tokens))
# " ".join(clean_tokens) = 리스트의 단어들을 공백으로 다시 이어붙여 하나의 긴 문장으로 만듦.
#                          예: ['배송','빠름'] → "배송 빠름"
#   (kkma가 리스트가 아니라 문자열을 입력으로 받기 때문에 이렇게 합침)
#
# kkma.pos(...) = 꼬꼬마(Kkma) 분석기가 문장을 형태소로 쪼개고
#                 각 형태소에 품사 태그를 붙여줌.
#   pos = part of speech(품사)
#
# 결과 dev 의 생김새:
#   [('배송', 'NNG'), ('이', 'JKS'), ('빠르', 'VA'), ...]
#   → (단어, 품사) 짝들의 리스트.
#     NNG = 일반명사, JKS = 조사, VA = 형용사 ...

print(dev)
# 분석 결과를 화면에 출력해서 눈으로 확인.


# ───────────────────────────────────────────────
# [5] 그중에서 '일반명사(NNG)'만 골라내기
# ───────────────────────────────────────────────

nouns = [word for word, tag in dev if tag == 'NNG']
# for word, tag in dev = dev에서 짝을 하나씩 꺼내는데, ('배송','NNG') 같은 두 칸짜리라
#                        왼쪽은 word, 오른쪽은 tag 에 자동으로 나눠 담음.
# if tag == 'NNG'       = 품사가 일반명사인 것만 통과.
# word (맨 앞)          = 통과한 것 중 '단어'만 꺼내서 리스트에 담음 (품사 기호는 버림).
#
# 결과: nouns = ['배송', '포장', '도착', '배송', ...]  ← 명사들만 모임 (중복 포함)


# ───────────────────────────────────────────────
# [6] 명사들의 등장 횟수 세기 (빈도 분석)
# ───────────────────────────────────────────────

counter = Counter(nouns)
# nouns 안의 단어들이 각각 몇 번 나왔는지 세서 counter에 저장.
#   예: Counter({'배송': 12, '포장': 5, '도착': 3, ...})

print(counter.most_common(20))
# most_common(20) = 가장 많이 나온 순서대로 상위 20개를 뽑아서 출력.
#   결과 예: [('배송', 12), ('포장', 5), ('도착', 3), ...]
#   → (단어, 횟수) 짝의 리스트로, 빈도가 높은 순으로 정렬됨.
```

2. okt 사용하기
``` python
import re
from collections import Counter
from konlpy.tag import Okt   # ① 꼬꼬마 대신 Okt 불러오기

okt = Okt()                  # ② Okt 분석기 준비

pattern = re.compile(r'[0-9a-zA-Z\W]+|[ㅋㅎㅍㅇ]+')

nouns = []   # 명사를 모을 리스트

# 리뷰를 하나씩 꺼내서 처리
for doc in deliver_df['리뷰']:
    # 1) 단어로 쪼개고 청소
    clean_tokens = [
        cleaned
        for token in doc.split()
        if (cleaned := pattern.sub('', token))
    ]

    if not clean_tokens:      # 남은 단어 없으면 건너뜀
        continue

    # 2) 청소된 단어들을 합쳐서 '명사만' 바로 추출
    nouns += okt.nouns(" ".join(clean_tokens))   # ③ 핵심: .nouns()

# 3) 빈도 분석
counter = Counter(nouns)
print(counter.most_common(20))
```

### 6. N-gram
- 연속적인 단어의 의미 보존을 위해 처리하는 방법
- 1-gram(유니그램): 한 단어(글자)를 토큰으로 등록
- 2-gram(바이그램): 두 단어(글자)를 토큰으로 등록
- 3-gram(트라이그램): 세 단어(글자)를 토큰으로 등록
- 너무 과도한 n-gram을 쓸 경우 토큰화의 의미 감소

``` python
from nltk.util import ngrams

# 토큰화 된 예시 데이터 준비
sample_tokens = ['오늘', '저녁에', '무엇을', '먹을까', '?']

# 바이그램
bi_result = list(ngrams(sample_tokens,2))

# 트라이그램
tri_result = list(ngrams(sample_tokens, 3))
```

#### N-gram을 적용해서 빈도 분석하기
- kiwi 활용
- 유니, 바이, 트라이 적용 후 빈도 측정

``` python
clean_reviews = [] # n-gram이 적용된 데이터가 담길 리스트

for doc in deliver_df['리뷰']: # 배송 리뷰 반복 처리
    morphs = Kiwi.tokenize(doc) # 형태소 분석
    morphs_tokens = [] # 일반명사, 동사, 형용사만 필터링 하는 리스트
    for i in morphs : # 토큰 반복 처리
        if i.tag in ['NNG', 'VV', 'VA'] : # 조건 처리
            morphs_tokens.append(i.form) # 토큰 추가

    clean_reviews.append(morphs_tokens) # 전체 리스트에 추가
```

``` python
n_gram_clean_reviews = [] # n-gram이 적용된 리뷰를 담을 리스트

for doc in clean_reviews : # 클렌징 된 전체데이터를 반복
  temp_tokens = [] # 하나의 리뷰에 대한 중간결과 리스트 <- (유니그램 토큰, 바이그램 토큰)
  bi_tokens = list(ngrams(doc,2)) # 바이그램 생성
  tri_tokens = list(ngrams(doc,3)) # 트라이그램 생성
  # 유니그램 데이터 추가
  temp_tokens = temp_tokens + doc
  # 바이그램 데이터 추가
  for i in bi_tokens :
    temp_tokens.append(" ".join(i)) # 두 단어를 하나의 문자열로 결합하여 추가
  # 트라이그램 데이터 추가
  for i in tri_tokens :
    temp_tokens.append(" ".join(i)) # 두 단어를 하나의 문자열로 결합하여 추가
  n_gram_clean_reviews.append(temp_tokens) # 전체리스트에 추가 
```
