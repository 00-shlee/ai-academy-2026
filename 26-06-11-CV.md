## 원본 이미지에 이진화 변형 이미지 대입

``` python
import cv2
```

``` python
apple = cv2.imread('./image/apple.jpg', cv2.IMREAD_GRAYSCALE)

height, width = apple.shape # (612, 612) => 높이, 너비

# 중앙 좌표값 구하기 (높이, 너비 // 2) => 소수점이 나올 수 있으므로 몫만 구함
center_x, center_y = width // 2, height // 2
# print(center_x, center_y) # (306, 306)

# 중앙 좌표 기준 x,y값의 절반 값 구하기
center_half_x, center_half_y = center_x // 2, center_y // 2
# print(center_half_x, 

# 사각형 영역 좌표 구하기(좌상단, 우하단)
# 이미지 좌표계는 좌측 상단이 원점(0, 0)
# 오른쪽으로 갈수록 x값이 커지고, 아래로 갈수록 y값이 커짐

# 중심 x좌표에서 절반 너비만큼 왼쪽으로 이동한 값 → 사각형의 왼쪽 x좌표
top_left_x = center_x - center_half_x 

# 중심 y좌표에서 절반 높이만큼 위쪽으로 이동한 값 → 사각형의 위쪽 y좌표
top_left_y = center_y - center_half_y

# 중심 x좌표에서 절반 너비만큼 오른쪽으로 이동한 값 → 사각형의 오른쪽 x좌표
btm_right_x = center_x + center_half_x

# 중심 y좌표에서 절반 높이만큼 아래쪽으로 이동한 값 → 사각형의 아래쪽 y좌표
btm_right_y = center_y + center_half_y


# 이미지 배열은 [y범위, x범위] = [행 범위, 열 범위] 순서로 접근함
# 계산한 좌표를 이용해 원본 이미지에서 중앙 사각형 영역만 잘라내기
rect_region = apple[top_left_y:btm_right_y, top_left_x:btm_right_x]

# 잘라낸 사각형 영역의 크기 확인
print(rect_region.shape)


# 이진화
# rect_region 영역을 흑백(0 또는 255)으로 변환
# THRESH_BINARY: 임계값 기준으로 픽셀을 검정/흰색으로 나눔
# THRESH_OTSU: 적절한 임계값을 자동으로 계산해 줌
thres, apple_bin = cv2.threshold(
    rect_region, 
    0, 
    255, 
    cv2.THRESH_BINARY | cv2.THRESH_OTSU
)


# 원본 이미지를 직접 수정하지 않기 위해 복사본 생성
apple_copy = apple.copy()

# 원본 이미지 복사본의 중앙 사각형 영역에 이진화된 이미지를 대입
# 즉, 전체 이미지는 유지하고 중앙 영역만 이진화 결과로 바꿈
apple_copy[top_left_y:btm_right_y, top_left_x:btm_right_x] = apple_bin


# Otsu 알고리즘이 자동으로 찾은 임계값 출력
print(thres)

# 중앙 영역만 이진화된 사과 이미지를 회색조로 시각화
plt.imshow(apple_copy, cmap='gray')
plt.show()
```

## 이미지 블렌딩
``` python
img1 = cv2.imread('./image/ryan.jpg')
img2 = cv2. imread('./image/apple.jpg')

print(img1.shape)
print(img2.shape)

img1 = cv2.cvtColor(img1, cv2.COLOR_BGR2RGB)
img2 = cv2.cvtColor(img2, cv2.COLOR_BGR2RGB)

# => 블렌딩(addWeighted) => 두 이미지의 사이즈가 같아야 함
# 두 이미지 사이즈 동일화
img1 = cv2.resize(img1, (300, 300))
img2 = cv2.resize(img2, (300, 300))

# cv2.addWeighted(src1, alpha, src2, beta, gamma)
# => 결과 이미지 = src1 * alpha + src2 * beta + gamma
image = cv2.addWeighted( img1, 0.5, img2, 0.5, 0) # 0 ~ 1 사이의 값을 사용함

import matplotlib.pyplot as plt
plt.subplot(1, 3, 1)
plt.imshow(img1)
plt.subplot(1, 3, 2)
plt.imshow(img2)
plt.subplot(1, 3, 3)
plt.imshow(image)
plt.show()
```

``` python
cap = cv2.VideoCapture('./image/dogs.mp4')

while True:
    ch, frame = cap.read() # 성공 여부, 프레임
    # ch -> True(성공), False(실패)

    if not ch: # 프레임을 읽어오지 못했을 때
        print('동영상 재생 끝 or 읽기 실패')
        break

    # 프레임을 읽어온 경우
    cv2.imshow('pets', frame)
    key = cv2.waitKey(1)
    if key == ord('q'): #ord: 문자를 숫자로 변환해 주는 기능
        print("사용자 요청으로 영상 종료")
        break

cap.release() # cap 자원 해제
cv2.destroyAllWindows()
```
