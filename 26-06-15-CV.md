# YOLO 기초

## 1. 객체 탐지 (Object Detection)

컴퓨터 비전 기술 중 하나로, 이미지나 영상 속에서 **<U>특정 객체의 종류(what)와 위치(where)를 찾아내는 기술</U>**, 단순히 이미지를 분류하는 것을 넘어, 객체 주변에 경계 상자(bounding box)를 그리고 어떤 객체인지 레이블(label)을 붙여주는 것이 핵심

### 이미지 분류와의 차이점

- **Classification(분류):** 여러 물체에 대해 어떤 물체인지 클래스를 분류하는 문제
- **Localization(위치 파악):** 물체가 어디 있는지 박스를 통해 위치 정보를 나타내는 문제
- **Object Detection(객체탐지):** Classification과 Localization을 동시에 진행하는 것

## 2. YOLO
이미지나 영상 속에서 물체(객체)의 위치와 종류를 '단 한 번만 보고'
한 번에 찾아내는 딥러닝 기반의 객체 탐지(Object Detection) 모델

**자세한 설명은 강의 중 정리 내용 참고**

``` python
# 파일 이름, 디렉토리 정보 확인 등 os관련된 명령을 수행하는 라이브러리 불러오기

import os

# 각 경로에 있는 파일명(리스트) 저장

apple_pie_file_names = os.listdir('./data/food/apple_pie')

# 각 음식별 파일 갯수 확인

apple_pie_file_names = os.listdir('./data/food/apple_pie')
bibimbap_file_names = os.listdir('./data/food/bibimbap')
caesar_salad_file_names = os.listdir('./data/food/caesar_salad')
gyoza_file_names = os.listdir('./data/food/gyoza')
takoyaki_file_names = os.listdir('./data/food/takoyaki')

print(
    len(apple_pie_file_names),
    len(bibimbap_file_names),
    len(caesar_salad_file_names),
    len(gyoza_file_names),
    len(takoyaki_file_names)
)

# 폴더생성하기 (data/food_for_yolo, food_for_yolo/train, food_for_yolo/test, food_for_yolo/val), 기존에 만들어진 폴더가 없을 경우에만 생성

# food_for_yolo 폴더 생성

if not os.path.exists('./data/food_for_yolo'): # 해당 폴더 존재 여부 확인 후 없을 때 생성
    os.mkdir('./data/food_for_yolo') # 생성

if not os.path.exists('./data/food_for_yolo/train'): # 해당 폴더 존재 여부 확인 후 없을 때 생성
    os.mkdir('./data/food_for_yolo/train')

if not os.path.exists('./data/food_for_yolo/test'): # 해당 폴더 존재 여부 확인 후 없을 때 생성
    os.mkdir('./data/food_for_yolo/test')

if not os.path.exists('./data/food_for_yolo/val'): # 해당 폴더 존재 여부 확인 후 없을 때 생성
    os.mkdir('./data/food_for_yolo/val')

# 하위폴더 생성하기(apple_pie, bibimbap, caesar_sald, gyoza, takoyaki)

food_list = ['apple_pie', 'bibimbap', 'caesar_salad', 'gyoza', 'takoyaki']
folder_list = ['./data/food_for_yolo/train', './data/food_for_yolo/test', './data/food_for_yolo/val']

for folder in folder_list:
    for food in food_list:
            if not os.path.exists(folder+ "/"+food): # 해당 폴더 존재 여부 확인 후 없을 때 생성
                os.mkdir(folder+ "/"+food)


# shutil 모듈불러오기 : 파일에 대한 고수준의 연산을 제공함

import shutil

# train set 사진 이동하기 (640장, 첫번째 음식)

for img_name in apple_pie_file_names[:640]:
    shutil.move('./data/food/apple_pie/'+img_name,
                './data/food_for_yolo/train/apple_pie/'+img_name)

# val set 사진 이동하기 (160장, 첫번째 음식)

for img_name in apple_pie_file_names[640:800]:
    shutil.move('./data/food/apple_pie/'+img_name,
                './data/food_for_yolo/val/apple_pie/'+img_name)

# test set 사진 이동하기 (200장, 첫번째 음식)

for img_name in apple_pie_file_names[800:]:
    shutil.move('./data/food/apple_pie/'+img_name,
                './data/food_for_yolo/test/apple_pie/'+img_name)


# 나머지 음식 이미지 파일 옮기기
food_file_name_list = [bibimbap_file_names, caesar_salad_file_names, gyoza_file_names, takoyaki_file_names]

for food_name, food_file_name in zip(food_list[1:], food_file_name_list):
    for img_name in food_file_name[:640]:
        shutil.move('./data/food/'+food_name+'/'+img_name,
                    './data/food_for_yolo/train/'+food_name+'/'+img_name)
    
    for img_name in food_file_name[640:800]:
        shutil.move('./data/food/'+food_name+'/'+img_name,
                    './data/food_for_yolo/val/'+food_name+'/'+img_name)
    
    for img_name in food_file_name[800:]:
        shutil.move('./data/food/'+food_name+'/'+img_name,
                    './data/food_for_yolo/test/'+food_name+'/'+img_name)

# 파일 이동여부 확인

train_gyoza = os.listdir('./data/food_for_yolo/train/gyoza')
val_gyoza = os.listdir('./data/food_for_yolo/val/gyoza')
test_gyoza = os.listdir('./data/food_for_yolo/test/gyoza')

print(len(train_gyoza), len(val_gyoza), len(test_gyoza))
```

``` python
#YOLO 모델 불러오기

from ultralytics import YOLO

# 미리 학습된 모델 가중치 파일을 이용해 YOLO 모델 객체 생성

%pwd

# yolo11의 nano 크기의 분류 모델 => 미리 학습된 모델
model = YOLO('./model/yolo11n-cls.pt')

# 모델 학습 실행

model.train(
    data='./data/food_for_yolo', 
    epochs=10, # 학습 횟수
    imgsz=224) # 224 * 224 크기로 조정하여 학습

# 모델 검증하기

model.val()
```
